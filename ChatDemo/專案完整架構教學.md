### 2.1 Contracts（介面層）

**IMemberService.cs**
```csharp
/// <summary>會員服務介面</summary>
public interface IMemberService {
    Task<bool> RegisterAsync(Member member); // 註冊會員
```

### 2.3 Application（業務邏輯層）

**MemberApplicationService.cs**
```csharp
/// <summary>會員應用服務，處理業務邏輯</summary>
public class MemberApplicationService {
    private readonly IMemberService _memberService;
    public MemberApplicationService(IMemberService memberService) {
        _memberService = memberService;
    }

    // 註冊邏輯：驗證資料、檢查重複、呼叫服務
    public async Task<ApplicationResult<bool>> RegisterMemberAsync(Member member) {
        if (string.IsNullOrWhiteSpace(member.Username) || string.IsNullOrWhiteSpace(member.Password))
            return ApplicationResult<bool>.Failure("用戶名和密碼不能為空");
        if (await _memberService.UsernameExistsAsync(member.Username))
            return ApplicationResult<bool>.Failure("用戶名已存在");
        var result = await _memberService.RegisterAsync(member);
        return result ? ApplicationResult<bool>.Success(true, "註冊成功") : ApplicationResult<bool>.Failure("註冊失敗");
    }

    // 登入邏輯
    public async Task<ApplicationResult<Member>> LoginMemberAsync(string username, string password) {
        if (string.IsNullOrWhiteSpace(username) || string.IsNullOrWhiteSpace(password))
            return ApplicationResult<Member>.Failure("用戶名和密碼不能為空");
        var member = await _memberService.LoginAsync(username, password);
        return member != null ? ApplicationResult<Member>.Success(member, "登入成功") : ApplicationResult<Member>.Failure("用戶名或密碼不正確");
    }
}
```

### 2.4 Implementation（實作層）

**MockMemberService.cs**
```csharp
/// <summary>Mock 會員服務，僅用於測試</summary>
public class MockMemberService : IMemberService {
    private static readonly List<Member> _mockMembers = new() {
        new Member { Id = 1, Username = "testuser", Password = BCrypt.Net.BCrypt.HashPassword("123456"), Email = "testuser@example.com" },
        new Member { Id = 2, Username = "alice", Password = BCrypt.Net.BCrypt.HashPassword("alicepwd"), Email = "alice@example.com" }
    };
    public Task<bool> RegisterAsync(Member member) {
        if (_mockMembers.Any(m => m.Username == member.Username)) return Task.FromResult(false);
        member.Id = _mockMembers.Max(m => m.Id) + 1;
        member.Password = BCrypt.Net.BCrypt.HashPassword(member.Password);
        _mockMembers.Add(member);
        return Task.FromResult(true);
    }
    public Task<Member?> LoginAsync(string username, string password) {
        var member = _mockMembers.FirstOrDefault(m => m.Username == username);
        if (member != null && BCrypt.Net.BCrypt.Verify(password, member.Password))
            return Task.FromResult<Member?>(member);
        return Task.FromResult<Member?>(null);
    }
    public Task<bool> UsernameExistsAsync(string username) {
        return Task.FromResult(_mockMembers.Any(m => m.Username == username));
    }
}
```
> 真實服務（DapperMemberService）則會連資料庫，原理類似。

### 2.5 Api（控制器層）

**MemberController.cs**
```csharp
[Route("api/[controller]")]
[ApiController]
public class MemberController : ControllerBase {
    private readonly MemberApplicationService _memberApplicationService;
    public MemberController(MemberApplicationService memberApplicationService) {
        _memberApplicationService = memberApplicationService;
    }

    // 註冊 API
    [HttpPost("register")]
    public async Task<IActionResult> Register([FromBody] Member member) {
        var result = await _memberApplicationService.RegisterMemberAsync(member);
        return result.IsSuccess ? Ok(new { success = true, message = result.Message }) : BadRequest(new { success = false, message = result.Message });
    }

    // 登入 API
    [HttpPost("login")]
    public async Task<IActionResult> Login([FromBody] LoginRequest loginRequest) {
        var result = await _memberApplicationService.LoginMemberAsync(loginRequest.Username, loginRequest.Password);
        return result.IsSuccess ? Ok(new { success = true, message = result.Message, data = new { username = result.Data?.Username, email = result.Data?.Email } }) : Unauthorized(new { success = false, message = result.Message });
    }
}
public class LoginRequest {
    public string Username { get; set; } = string.Empty;
    public string Password { get; set; } = string.Empty;
}
```

### 3.2 Login.vue
```vue
<template>
  <div class="login-container" :style="{ backgroundImage: 'url(' + backgroundImage + ')', backgroundSize: 'cover', backgroundPosition: 'center' }">
    <h2>請先登入以使用聊天機器人</h2>
    <div class="login-box">
      <input v-model="username" placeholder="用戶名" />
      <input v-model="password" type="password" placeholder="密碼" />
      <button @click="login">登入</button>
      <p v-if="error">{{ error }}</p>
      <button @click="goToRegister">沒有帳戶？前往註冊</button>
    </div>
  </div>
</template>
<script setup>
import { ref } from 'vue';
import { useRouter } from 'vue-router';
import axios from 'axios';
import backgroundImage from '@/assets/background.gif';
import { useUserStore } from '@/stores/user';

const username = ref('');
const password = ref('');
const error = ref('');
const router = useRouter();
const userStore = useUserStore();

const login = async () => {
  if (!username.value || !password.value) {
    error.value = '請輸入用戶名和密碼！';
    return;
  }
  try {
    const response = await axios.post('http://localhost:5006/api/Member/login', {
      username: username.value,
      password: password.value,
      email: ''
    });
    if (response.data.success) {
      userStore.login({ username: username.value, password: password.value });
      router.push('/');
    } else {
      error.value = '帳號密碼錯誤';
    }
  } catch (err) {
    error.value = err.response ? err.response.data.message : '伺服器無回應，請稍後再試。';
  }
};
const goToRegister = () => { router.push('/register'); };
</script>
```

### 3.3 Pinia 狀態管理 user.js
```js
import { defineStore } from 'pinia';
import { ref } from 'vue';

export const useUserStore = defineStore('user', () => {
  const username = ref('');
  const password = ref('');
  const loggedIn = ref(false);

  function login(payload) {
    username.value = payload.username;
    password.value = payload.password;
    loggedIn.value = true;
  }
  function logout() {
    username.value = '';
    password.value = '';
    loggedIn.value = false;
  }
  return { username, password, loggedIn, login, logout };
});
```
